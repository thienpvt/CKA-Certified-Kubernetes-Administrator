    ```bash
    #!/bin/bash
    # cka-sim/tests/bin/kubectl — PATH-shadow stub for unit tests.
    #
    # Reads argv, fingerprints into a fixture path, and cats the matching JSON file.
    # Real kubectl is silent on the happy path; we mirror that.
    #
    # Inputs:
    #   $CKA_SIM_TEST_FIXTURES_DIR   — base dir (set by tests/run.sh)
    #   $CKA_SIM_TEST_CURRENT        — current case fixture sub-dir (set by case files)
    # Output:
    #   stdout = fixture file contents (raw JSON)
    #   stderr = errors only (unsupported argv shape, missing fixture)
    # Exit: 0 on success, 1 on missing fixture, 64 on unsupported verb.

    set -euo pipefail

    : "${CKA_SIM_TEST_FIXTURES_DIR:?CKA_SIM_TEST_FIXTURES_DIR must be set by tests/run.sh}"
    : "${CKA_SIM_TEST_CURRENT:?CKA_SIM_TEST_CURRENT must be set by the test case (e.g. default-sa-used/hit)}"

    fixture_path="$CKA_SIM_TEST_FIXTURES_DIR/$CKA_SIM_TEST_CURRENT.json"

    verb="${1:-}"
    case "$verb" in
      get)
        # kubectl get <resource> [<name>] [-n <ns>] [-o json|jsonpath=...]
        # Simple dispatcher: cat the fixture; tests pre-set CKA_SIM_TEST_CURRENT to point
        # at the right hit/miss/benign file.
        if [[ ! -f "$fixture_path" ]]; then
          printf 'kubectl-stub: fixture not found: %s\n' "$fixture_path" >&2
          exit 1
        fi
        # If args contain `-o jsonpath=...`, evaluate against the fixture JSON via jq;
        # otherwise dump the raw JSON.
        local_jsonpath=""
        local_use_name_only=0
        for arg in "$@"; do
          case "$arg" in
            -o)
              ;;
            jsonpath=*)
              local_jsonpath="${arg#jsonpath=}"
              ;;
            -o=jsonpath=*)
              local_jsonpath="${arg#-o=jsonpath=}"
              ;;
            json|-o=json)
              ;;
            name|-o=name)
              local_use_name_only=1
              ;;
          esac
        done
        if [[ -n "$local_jsonpath" ]]; then
          # Translate kubectl jsonpath {.foo.bar} → jq '.foo.bar // ""'.
          # Strip leading/trailing braces; handle the common shapes used by detectors:
          #   {.spec.serviceAccountName}
          #   {.status.phase}
          #   {.status.conditions[?(@.type=="Ready")].status}
          #   {.subsets[*].addresses[*].ip}
          jpath="${local_jsonpath#\{}"
          jpath="${jpath%\}}"
          # The conditions[?()] selector is the only kubectl-specific shape we need to translate.
          # Convert kubectl's [?(@.type=="Ready")] → [(.type=="Ready")].
          jq_expr=$(echo "$jpath" | sed -E '
            s/\[\?\(@\.([a-zA-Z]+)=="([^"]+)"\)\]/[] | select(.\1=="\2")/g
            s/\[\*\]/[]/g
          ')
          # Compose: . then jq_expr; default empty string when null.
          # `|| true` protects the stub's `set -euo pipefail` from aborting when jq exits non-zero
          # on a malformed/unsupported jsonpath — an empty stdout is the desired "field not found" signal.
          { jq -r "$jq_expr // \"\"" "$fixture_path" 2>/dev/null | tr '\n' ' ' | sed 's/[[:space:]]*$//'; } || true
          printf '\n'
        elif (( local_use_name_only == 1 )); then
          # `-o name` returns one resource per line, like "<kind>/<name>"
          jq -r '.kind + "/" + .metadata.name' "$fixture_path" 2>/dev/null
        else
          cat "$fixture_path"
        fi
        ;;
      auth)
        # kubectl auth can-i <verb> <resource> [-n <ns>] [--as <user>]
        if [[ "${2:-}" != "can-i" ]]; then
          printf 'kubectl-stub: unsupported auth subcommand: %s\n' "${2:-}" >&2
          exit 64
        fi
        # Fixture: a single token "yes" or "no" on stdout.
        if [[ ! -f "$fixture_path" ]]; then
          printf 'kubectl-stub: fixture not found: %s\n' "$fixture_path" >&2
          exit 1
        fi
        cat "$fixture_path"
        ;;
      exec)
        # kubectl exec -n NS POD -- CMD ... — fixture decides exit code via filename suffix.
        # Convention: fixture file content = exit code (e.g. "0\n" for success, "1\n" for fail).
        if [[ ! -f "$fixture_path" ]]; then
          printf 'kubectl-stub: fixture not found: %s\n' "$fixture_path" >&2
          exit 1
        fi
        rc=$(head -1 "$fixture_path")
        exit "${rc:-0}"
        ;;
      describe)
        if [[ ! -f "$fixture_path" ]]; then
          printf 'kubectl-stub: fixture not found: %s\n' "$fixture_path" >&2
          exit 1
        fi
        cat "$fixture_path"
        ;;
      *)
        printf 'kubectl-stub: unsupported verb %q (set CKA_SIM_TEST_CURRENT to a fixture; see tests/fixtures/)\n' "$verb" >&2
        exit 64
        ;;
    esac
    ```

    ```bash
    #!/bin/bash
    # cka-sim/tests/lib/assert.sh — micro-assertions for bash unit cases.
    # Sourced by every cka-sim/tests/cases/*.sh.
    # Helpers return 0 on pass, 1 on fail; never `die`. The case file's caller
    # (run.sh) inspects return codes to aggregate pass/fail.

    : "${CKA_SIM_ROOT:?CKA_SIM_ROOT must be set}"
    # shellcheck disable=SC1091
    [[ -z "${RED+x}" ]] && source "$CKA_SIM_ROOT/lib/colors.sh"

    expect_eq() {
      local actual="${1:-}" expected="${2:-}" msg="${3:-expect_eq}"
      if [[ "$actual" == "$expected" ]]; then
        printf '%s  ✓ %s%s\n' "$GREEN" "$msg" "$NC" >&2
        return 0
      fi
      printf '%s  ✗ %s — expected %q got %q%s\n' "$RED" "$msg" "$expected" "$actual" "$NC" >&2
      return 1
    }

    expect_empty() {
      local actual="${1:-}" msg="${2:-expect_empty}"
      if [[ -z "$actual" ]]; then
        printf '%s  ✓ %s%s\n' "$GREEN" "$NC" "$msg" >&2
        return 0
      fi
      printf '%s  ✗ %s — expected empty, got %q%s\n' "$RED" "$msg" "$actual" "$NC" >&2
      return 1
    }

    expect_contains() {
      local haystack="${1:-}" needle="${2:-}" msg="${3:-expect_contains}"
      if [[ "$haystack" == *"$needle"* ]]; then
        printf '%s  ✓ %s%s\n' "$GREEN" "$NC" "$msg" >&2
        return 0
      fi
      printf '%s  ✗ %s — %q does not contain %q%s\n' "$RED" "$msg" "$haystack" "$needle" "$NC" >&2
      return 1
    }

    expect_match() {
      local actual="${1:-}" pattern="${2:-}" msg="${3:-expect_match}"
      if [[ "$actual" =~ $pattern ]]; then
        printf '%s  ✓ %s%s\n' "$GREEN" "$NC" "$msg" >&2
        return 0
      fi
      printf '%s  ✗ %s — %q does not match /%s/%s\n' "$RED" "$msg" "$actual" "$pattern" "$NC" >&2
      return 1
    }
    ```

    No `set -e/-u/-o pipefail` at the top — this is a sourceable lib. The `:?` defensive check on `CKA_SIM_ROOT` is the only gate.

    A. Create `cka-sim/tests/run.sh` (the case-walker; uses `set -uo pipefail`, NOT `-e`, so all cases run):

    ```bash
    #!/bin/bash
    # cka-sim/tests/run.sh — bash unit-test runner for traps.sh + grade.sh.
    # Walks cka-sim/tests/cases/*.sh, sources each in a subshell, aggregates pass/fail.
    # Local: bash cka-sim/tests/run.sh
    # CI: invoked by cka-sim/scripts/test.sh.

    set -uo pipefail   # NOT -e: continue past failing cases to aggregate results

    CKA_SIM_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
    export CKA_SIM_ROOT

    # PATH-shadow real kubectl with the test stub.
    export PATH="$CKA_SIM_ROOT/tests/bin:$PATH"

    # Fixtures dir — case files set CKA_SIM_TEST_CURRENT to a sub-path under here.
    export CKA_SIM_TEST_FIXTURES_DIR="$CKA_SIM_ROOT/tests/fixtures"

    # shellcheck source=../lib/colors.sh disable=SC1091
    source "$CKA_SIM_ROOT/lib/colors.sh"
    # shellcheck source=../lib/log.sh disable=SC1091
    source "$CKA_SIM_ROOT/lib/log.sh"

    header "cka-sim bash unit tests"

    cases_dir="$CKA_SIM_ROOT/tests/cases"
    if [[ ! -d "$cases_dir" ]]; then
      warn "no cases directory at $cases_dir — nothing to run (treat as success during scaffold)"
      exit 0
    fi

    total=0
    failed=0

    while IFS= read -r -d '' case_file; do
      total=$(( total + 1 ))
      header "$(basename "$case_file" .sh)"
      # Run case in a subshell so its set -u/-o pipefail / variable leaks don't poison run.sh.
      if ( source "$case_file" ); then
        ok "case passed: $(basename "$case_file" .sh)"
      else
        err "case failed (rc=$?): $(basename "$case_file" .sh)"
        failed=$(( failed + 1 ))
      fi
    done < <(find "$cases_dir" -name '*.sh' -print0 | sort -z)

    printf '\n' >&2
    if (( total == 0 )); then
      warn "no test cases found in $cases_dir — treat as success during scaffold"
      exit 0
    fi
    if (( failed == 0 )); then
      ok "all $total case(s) passed"
      exit 0
    else
      err "$failed of $total case(s) failed"
      exit 1
    fi
    ```

    Note: when `cases_dir` is empty (this plan ships the harness BEFORE plan 02-04 ships the cases), run.sh exits 0 with a warning. This is intentional so plan 02-03 closes green; plan 02-04 will populate `cases/` and the run will start counting pass/fail.

    ```bash
          fi

    }

      v="${v#\'}"; v="${v%\'}"

        last=$(( ${#current_ref_targets[@]} - 1 ))

        fi
    ```

    ```
    # storage/06-pvc-mount-pod

    **Domain:** Storage  |  **Estimated time:** 7 minutes

    A `PersistentVolumeClaim` named `q06-data` is already Bound in `${CKA_SIM_LAB_NS}`. Its backing volume contains a pre-written file at `/data/marker`.

    ## Tasks

    1. Create a `Deployment` named `q06-reader` in `${CKA_SIM_LAB_NS}`.
    2. Each pod in the Deployment must mount `q06-data` **read-only** under `/data`.
    3. Use image `nginx:1.27`, `replicas: 1`, and `resources.requests` set to a non-zero cpu + memory.

    ## Constraints

    - Do NOT modify `q06-data` or its PV.
    - Do NOT set `readOnly: false` or omit it.

    ## Verify yourself

    ```
    kubectl get deploy q06-reader -n ${CKA_SIM_LAB_NS}
    kubectl exec -n ${CKA_SIM_LAB_NS} deploy/q06-reader -- cat /data/marker
    ```
    ```

    **setup.sh** (chmod +x):
    ```
    #!/bin/bash
    set -euo pipefail
    : "${CKA_SIM_LAB_NS:?}"
    : "${CKA_SIM_ROOT:?}"
    source "$CKA_SIM_ROOT/lib/setup.sh"

    cka_sim::setup::ensure_lab_ns "$CKA_SIM_LAB_NS" storage storage-pvc-mount-pod
    cka_sim::setup::wait_for_ns_active "$CKA_SIM_LAB_NS" storage storage-pvc-mount-pod 120

    # Hostpath PV with nodeAffinity (NOT the trap here -- correctly pinned).
    cka_sim::setup::seed_pv_hostpath q06-data-pv 1Gi ReadWriteOnce Retain /tmp/q06-data kubernetes.io/hostname
    kubectl patch pv q06-data-pv --type=json -p='[{"op":"add","path":"/spec/storageClassName","value":"manual"}]' || true

    # Pre-bound PVC
    kubectl apply -f - <<EOF
    apiVersion: v1
    kind: PersistentVolumeClaim
    metadata:
      name: q06-data
      namespace: ${CKA_SIM_LAB_NS}
    spec:
      accessModes: [ReadWriteOnce]
      resources: {requests: {storage: 500Mi}}
      storageClassName: manual
    EOF

    kubectl wait --for=jsonpath='{.status.phase}'=Bound pvc/q06-data -n "$CKA_SIM_LAB_NS" --timeout=60s 2>/dev/null || true

    # Writer pod: runs once, writes marker, exits. restartPolicy: OnFailure keeps it from restarting after success.
    kubectl apply -f - <<EOF
    apiVersion: v1
    kind: Pod
    metadata:
      name: q06-writer
      namespace: ${CKA_SIM_LAB_NS}
    spec:
      restartPolicy: OnFailure
      containers:
        - name: writer
          image: busybox:1.36
          command: ["sh","-c","echo q06-marker > /data/marker"]
          volumeMounts: [{name: data, mountPath: /data}]
      volumes:
        - name: data
          persistentVolumeClaim: {claimName: q06-data}
    EOF

    # Wait for writer completion.
    kubectl wait --for=jsonpath='{.status.phase}'=Succeeded pod/q06-writer -n "$CKA_SIM_LAB_NS" --timeout=60s 2>/dev/null || true
    ```

    set -uo pipefail
    : "${CKA_SIM_LAB_NS:?}"
    : "${CKA_SIM_ROOT:?}"
    source "$CKA_SIM_ROOT/lib/grade.sh"
    source "$CKA_SIM_ROOT/lib/traps.sh"

    kubectl wait --for=condition=Available deployment/q06-reader -n "$CKA_SIM_LAB_NS" --timeout=60s 2>/dev/null || true

    cka_sim::grade::assert_resource_exists deployment q06-reader -n "$CKA_SIM_LAB_NS"
    cka_sim::grade::assert_field_eq deployment q06-reader \
      '{.spec.template.spec.volumes[0].persistentVolumeClaim.claimName}' \
      'q06-data' -n "$CKA_SIM_LAB_NS"
    cka_sim::grade::assert_field_eq deployment q06-reader \
      '{.spec.template.spec.containers[0].volumeMounts[0].readOnly}' \
      'true' -n "$CKA_SIM_LAB_NS"

    # Behavioural: exec into the pod and cat the marker.
    marker=$(kubectl exec -n "$CKA_SIM_LAB_NS" deployment/q06-reader -- cat /data/marker 2>/dev/null | tr -d '\n\r ')
    CKA_SIM_GRADE_TOTAL=$(( CKA_SIM_GRADE_TOTAL + 1 ))
    if [[ "$marker" == "q06-marker" ]]; then
      CKA_SIM_GRADE_PASSED=$(( CKA_SIM_GRADE_PASSED + 1 ))
      ok "exec probe: /data/marker == 'q06-marker'"
    else
      err "exec probe: /data/marker = '$marker' (expected 'q06-marker')"

    # Trap detector: check default SA on any deployment pod
    pod=$(kubectl get pod -n "$CKA_SIM_LAB_NS" -l app=q06-reader -o jsonpath='{.items[0].metadata.name}' 2>/dev/null || echo "")
